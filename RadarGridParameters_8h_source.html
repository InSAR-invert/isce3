325 validate()
326 {
327  std::string errstr = "";
328  if (numberAzimuthLooks() < 1)
329  {
330  errstr += "Azimuth looks should be an integer greater than or equal to 1. \n";
331  }
332 
333  if (numberRangeLooks() < 1)
334  {
335  errstr += "Range looks should be an integer greather than or equal to 1. \n";
336  }
337 
338  if (wavelength() <= 0.)
339  {
340  errstr += "Radar wavelength must be positive. \n";
341  }
342 
343  if (prf() <= 0.)
344  {
345  errstr += "Pulse Repetition Frequency must be positive. \n";
346  }
347 
348  if (startingRangeSingleLook() <= 0.)
349  {
350  errstr += "Starting Range must be positive. \n";
351  }
352 
353  if (rangePixelSpacingSingleLook() <= 0. )
354  {
355  errstr += "Slant range pixel spacing must be positive. \n";
356  }
357 
358  if (length() == 0)
359  {
360  errstr += "Radar Grid should have length of atleast 1. \n";
361  }
362 
363  if (width() == 0)
364  {
365  errstr += "Radar Grid should have width of atleast 1. \n";
366  }
367 
368  if (! errstr.empty())
369  {
370  throw isce::except::InvalidArgument(ISCE_SRCINFO(), errstr);
371  }

376 void
377 isce::product::RadarGridParameters::
378 lookSide(const std::string & inputLook) {
379  // Convert to lowercase
380  std::string look(inputLook);
381  std::for_each(look.begin(), look.end(), [](char & c) {
382  c = std::tolower(c);
383  });
384  // Validate look string before setting
385  if (look.compare("right") == 0) {
386  _lookSide = -1;
387  } else if (look.compare("left") == 0) {
388  _lookSide = 1;
389  } else {
390  pyre::journal::error_t error("isce.product.RadarGridParameters");
391  error
392  << pyre::journal::at(__HERE__)
393  << "Could not successfully set look direction. Not 'right' or 'left'."
394  << pyre::journal::endl;
395  }
