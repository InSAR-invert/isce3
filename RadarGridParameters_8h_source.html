287 void
288 isce::product::RadarGridParameters::
289 lookSide(const std::string & inputLook) {
290  // Convert to lowercase
291  std::string look(inputLook);
292  std::for_each(look.begin(), look.end(), [](char & c) {
293  c = std::tolower(c);
294  });
295  // Validate look string before setting
296  if (look.compare("right") == 0) {
297  _lookSide = -1;
298  } else if (look.compare("left") == 0) {
299  _lookSide = 1;
300  } else {
301  pyre::journal::error_t error("isce.product.RadarGridParameters");
302  error
303  << pyre::journal::at(__HERE__)
304  << "Could not successfully set look direction. Not 'right' or 'left'."
305  << pyre::journal::endl;
306  }
