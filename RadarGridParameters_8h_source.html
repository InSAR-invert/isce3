158  inline RadarGridParameters multilook(size_t azlooks, size_t rglooks) const
159  {
160  //Check for number of points on edge
161  if ((azlooks == 0) || (rglooks == 0))
162  {
163  std::string errstr = "Number of looks must be positive. " +
164  std::to_string(azlooks) + "Az x" +
165  std::to_string(rglooks) + "Rg looks requested.";
166  throw isce::except::OutOfRange(ISCE_SRCINFO(), errstr);
167  }
168 
169  //Currently implements the multilooking operation used in ISCE2
170  return RadarGridParameters( sensingTime(0.5 * (azlooks-1)),
171  wavelength(),
172  prf() / (1.0 * azlooks),
173  slantRange(0.5 * (rglooks-1)),
174  rangePixelSpacing() * rglooks,
175  lookSide(),
176  length()/azlooks,
177  width()/rglooks,
178  refEpoch());
179  }

352 void
353 isce::product::RadarGridParameters::
354 lookSide(const std::string & inputLook) {
355  // Convert to lowercase
356  std::string look(inputLook);
357  std::for_each(look.begin(), look.end(), [](char & c) {
358  c = std::tolower(c);
359  });
360  // Validate look string before setting
361  if (look.compare("right") == 0) {
362  _lookSide = -1;
363  } else if (look.compare("left") == 0) {
364  _lookSide = 1;
365  } else {
366  pyre::journal::error_t error("isce.product.RadarGridParameters");
367  error
368  << pyre::journal::at(__HERE__)
369  << "Could not successfully set look direction. Not 'right' or 'left'."
370  << pyre::journal::endl;
371  }
